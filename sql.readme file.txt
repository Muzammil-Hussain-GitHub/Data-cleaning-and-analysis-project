#How many cars does each seller have?
#What is the average price of their cars?
#Which years are their cars?

SELECT 
    s.seller_name,
    s.seller_type,
    COUNT(c.car_id) AS number_of_cars,
    MIN(c.year) AS oldest_car_year,
    MAX(c.price_pkr) AS highest_price
FROM used_carsinfo AS c
JOIN sellers_data AS s ON c.seller_id = s.seller_id
GROUP BY s.seller_id, s.seller_name, s.seller_type;

 
 
 #How many cars of each model each seller sells
 
SELECT 
    s.seller_name,
    c.model,
    COUNT(*) AS number_of_cars
FROM used_carsinfo c
JOIN sellers_data s ON c.seller_id = s.seller_id
GROUP BY s.seller_id, s.seller_name, c.model
ORDER BY s.seller_name, number_of_cars DESC;

#  per seller and per car make (manufacturer) how many cars they sold, and also how many different models within that make.
SELECT 
    s.seller_name,
    c.make,
    COUNT(*) AS total_cars_sold
FROM used_carsinfo c
JOIN sellers_data s ON c.seller_id = s.seller_id
GROUP BY s.seller_id, s.seller_name, c.make
ORDER BY s.seller_name, c.make;

# How many models were sold by each car company and the corresponding models
SELECT 
    c.make,
    c.model,
    COUNT(*) AS total_sold
FROM used_carsinfo c
GROUP BY c.make, c.model
ORDER BY total_sold DESC;

 #How many distinct models exist for each car make (e.g., Honda has 3 models)
 SELECT 
    c.make,
    COUNT(DISTINCT c.model) AS number_of_distinct_models
FROM used_carsinfo c
GROUP BY c.make
ORDER BY number_of_distinct_models DESC;

# Add a column that labels price as High, Mid or Low
SELECT 
    Make, Model, Price_PKR,
    CASE 
        WHEN Price_PKR >= 4000000 THEN 'High'
        WHEN Price_PKR BETWEEN 2000000 AND 3999999 THEN 'Mid'
        ELSE 'Low'
    END AS Price_Category
FROM used_carsinfo ;

# Rank cars by price within each Make using window function

SELECT 
    Make, Model, Price_PKR,
    RANK() OVER (PARTITION BY Make ORDER BY Price_PKR DESC) AS Price_Rank
FROM used_carsinfo ;


# to get the top 3 cars per brand 
WITH Top3PerBrand AS (
    SELECT 
        Make,
        Model,
        Price_PKR,
        ROW_NUMBER() OVER (PARTITION BY Make ORDER BY Price_PKR DESC) AS rn
    FROM used_carsinfo   -- ✅ correct table name
)
SELECT Make, Model, Price_PKR, rn AS Price_Rank
FROM Top3PerBrand
WHERE rn <= 3
ORDER BY Make, Price_PKR DESC;

#incase we want the name of the seller and thier id too 
 
 WITH Top3PerBrand AS (
    SELECT 
        c.Make,
        c.Model,
        c.Price_PKR,
        s.seller_id,
        s.seller_name,          -- use the exact column name from seller_data
        ROW_NUMBER() OVER (PARTITION BY c.Make ORDER BY c.Price_PKR DESC) AS rn
    FROM used_carsinfo c
    JOIN sellers_data s ON c.seller_id = s.seller_id
)
SELECT 
    Make,
    Model,
    Price_PKR,
    seller_id,
    seller_name,
    rn AS Price_Rank
FROM Top3PerBrand
WHERE rn <= 3
ORDER BY Make, Price_PKR DESC;
